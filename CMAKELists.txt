cmake_minimum_required(VERSION 3.28)
project(eXpp CXX)

set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

if(NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Release)
endif()

set(CMAKE_CXX_FLAGS "-Wall -Wextra")
set(CMAKE_CXX_FLAGS_DEBUG "-g")
set(CMAKE_CXX_FLAGS_RELEASE "-O3")

if (CMAKE_SYSTEM_NAME STREQUAL "Linux" AND EXISTS "/etc/cros_release")
    add_definitions(-DCHROMEOS_BUILD)
endif()

set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY "${CMAKE_CURRENT_LIST_DIR}/build/lib")
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY "${CMAKE_CURRENT_LIST_DIR}/build/lib")
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_CURRENT_LIST_DIR}/build/bin")


include_directories(
    include/
)



add_library(expp_base STATIC)
target_sources(expp_base PUBLIC
    src/core/exception.cpp
)

add_library(coroutine INTERFACE)
target_sources(coroutine PUBLIC 
    src/core/coroutine/coroutine.cpp
)
add_executable(coroutine_test test/core/coroutine/coroutine.cpp)
target_link_libraries(coroutine_test PRIVATE coroutine)


add_library(memory INTERFACE)
target_sources(memory PUBLIC 
    src/core/memory/pointer.cpp
    src/core/memory/allocator.cpp
    src/core/memory/allocation.cpp
    src/core/memory/utils.cpp
)
add_executable(memory_test test/core/memory/memory.cpp)
target_link_libraries(memory_test PRIVATE memory expp_base)

add_library(context INTERFACE)
target_sources(context PUBLIC
    src/core/context/context.cpp
)
target_link_libraries(context INTERFACE memory)

add_executable(context_test test/core/context/context.cpp)
target_link_libraries(context_test PRIVATE context)


if(false)


add_executable(expp_ds_test test/core/ds/list.cpp)
target_link_libraries(expp_ds_test PRIVATE expp_ds)





add_library(stream STATIC)
target_sources(stream PUBLIC
    src/core/stream/stream.cpp
    src/core/stream/bitstream.cpp
)

add_executable(stream_test test/core/stream/bitstream.cpp)
target_link_libraries(stream_test PRIVATE stream)


add_library(image_png STATIC)
target_sources(image_png PUBLIC
    src/image/png/decoder.cpp
)


add_executable(image_png_test test/image/png/decoder.cpp)
target_link_libraries(image_png_test PRIVATE image_png stream)

endif()
