cmake_minimum_required(VERSION 3.28)
project(ModTest CXX)

set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

if (CMAKE_SYSTEM_NAME STREQUAL "Linux" AND EXISTS "/etc/cros_release")
    add_definitions(-DCHROMEOS_BUILD)
endif()

set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY "${CMAKE_CURRENT_LIST_DIR}/build/lib")
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY "${CMAKE_CURRENT_LIST_DIR}/build/lib")
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_CURRENT_LIST_DIR}/build/bin")


include_directories(
    include/
    src/
)



add_library(memory STATIC)
target_sources(memory PUBLIC 
    src/core/memory/allocator.cpp
    src/core/memory/allocation.cpp
    src/core/memory/memory.cpp
)

add_library(stream STATIC)
target_sources(stream PUBLIC
    src/core/stream/stream.cpp
    src/core/stream/bitstream.cpp
)

add_executable(stream_test test/core/stream/bitstream.cpp)
target_link_libraries(stream_test PRIVATE stream)


add_library(image_png STATIC)
target_sources(image_png PUBLIC
    src/image/png/decoder.cpp
)


add_executable(image_png_test test/image/png/decoder.cpp)
target_link_libraries(image_png_test PRIVATE image_png stream)
